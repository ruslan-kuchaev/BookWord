<script>

    function togglePassword() {
        const passwordInput = document.getElementById('password');
        const toggleButton = event.target;
        
        if (passwordInput.type === 'password') {
            passwordInput.type = 'text';
            toggleButton.textContent = 'Скрыть';
        } else {
            passwordInput.type = 'password';
            toggleButton.textContent = 'Показать';
        }
    }
    
    document.getElementById('password').addEventListener('input', function(e) {
        const password = e.target.value;
        
        const lengthReq = document.getElementById('length');
        if (password.length >= 8) {
            lengthReq.classList.add('valid');
            lengthReq.classList.remove('invalid');
            lengthReq.querySelector('i').className = 'fas fa-check-circle me-2 fa-xs text-success';
        } else {
            lengthReq.classList.add('invalid');
            lengthReq.classList.remove('valid');
            lengthReq.querySelector('i').className = 'fas fa-circle me-2 fa-xs text-muted';
        }
        
        const uppercaseReq = document.getElementById('uppercase');
        if (/[A-Z]/.test(password)) {
            uppercaseReq.classList.add('valid');
            uppercaseReq.classList.remove('invalid');
            uppercaseReq.querySelector('i').className = 'fas fa-check-circle me-2 fa-xs text-success';
        } else {
            uppercaseReq.classList.add('invalid');
            uppercaseReq.classList.remove('valid');
            uppercaseReq.querySelector('i').className = 'fas fa-circle me-2 fa-xs text-muted';
        }
        
        const lowercaseReq = document.getElementById('lowercase');
        if (/[a-z]/.test(password)) {
            lowercaseReq.classList.add('valid');
            lowercaseReq.classList.remove('invalid');
            lowercaseReq.querySelector('i').className = 'fas fa-check-circle me-2 fa-xs text-success';
        } else {
            lowercaseReq.classList.add('invalid');
            lowercaseReq.classList.remove('valid');
            lowercaseReq.querySelector('i').className = 'fas fa-circle me-2 fa-xs text-muted';
        }
        
        const specialReq = document.getElementById('special');
        if (/[$!%*?&]/.test(password)) {
            specialReq.classList.add('valid');
            specialReq.classList.remove('invalid');
            specialReq.querySelector('i').className = 'fas fa-check-circle me-2 fa-xs text-success';
        } else {
            specialReq.classList.add('invalid');
            specialReq.classList.remove('valid');
            specialReq.querySelector('i').className = 'fas fa-circle me-2 fa-xs text-muted';
        }
        
        const numberReq = document.getElementById('number');
        if (/[0-9]/.test(password)) {
            numberReq.classList.add('valid');
            numberReq.classList.remove('invalid');
            numberReq.querySelector('i').className = 'fas fa-check-circle me-2 fa-xs text-success';
        } else {
            numberReq.classList.add('invalid');
            numberReq.classList.remove('valid');
            numberReq.querySelector('i').className = 'fas fa-circle me-2 fa-xs text-muted';
        }
    });
</script>
